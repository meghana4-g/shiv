def add(a, b):
    """Function to add two numbers."""
    return a + b

def subtract(a, b):
    """Function to subtract b from a."""
    return a - b

# Example usage
num1 = 10
num2 = 5

# Adding two numbers
result_addition = add(num1, num2)
print(f"{num1} + {num2} = {result_addition}")

# Subtracting one number from another
result_subtraction = subtract(num1, num2)
print(f"{num1} - {num2} = {result_subtraction}")
